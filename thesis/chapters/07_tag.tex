\chapter{Temporal Attack Graph (\tagraph{})}
\label{ch:tag}

\section{Graph Structure}

The \tagraph{} $\mathcal{G} = (V, E, w)$ is a directed weighted graph:
\begin{itemize}
  \item \textbf{Attack nodes} $A_k$ ($k \in \{0,\ldots,7\}$): one per
    attack type; weight $= $ observed count; score $= $ EMA mean success.
  \item \textbf{Defense nodes} $D_j$ ($j \in \{0,\ldots,7\}$): one per
    countermeasure; weight $= $ deployment count; score $= $ EMA mean reward.
  \item \textbf{A--A edges} $(A_k \to A_{k'})$: weight $= $ co-occurrence
    count of $k$ immediately followed by $k'$ in the attack history.
  \item \textbf{A--D edges} $(A_k \to D_j)$: weight $= $ count of applying
    $D_j$ in response to $A_k$; attribute $\texttt{reward} = $ EMA defense reward.
\end{itemize}

All attributes are updated using EMA with decay $0.85$, giving recent
events higher weight than historical ones.

\section{Attack Sequence Prediction}

\subsection{Markov Chain Transition Probabilities}
\begin{equation}
  P(A_{k'} \mid A_k) = \frac{w(A_k \to A_{k'})}
    {\sum_{k''} w(A_k \to A_{k''})}
\end{equation}
Given recent attack history $\{k_{t-m}, \ldots, k_{t-1}\}$, the
most likely next attack is:
\begin{equation}
  k^* = \argmax_{k'} P(A_{k'} \mid A_{k_{t-1}})
\end{equation}
with confidence $c = P(A_{k^*} \mid A_{k_{t-1}})$.

\subsection{Proactive Defense Trigger}
\begin{equation}
  a^*_{\mathrm{pro}} = \begin{cases}
    \argmax_j \E[r_d \mid A_{k^*}, D_j] & \text{if } c \geq \tau_{\mathrm{pro}} \\
    \text{None} & \text{otherwise}
  \end{cases}
\end{equation}
where $\tau_{\mathrm{pro}} = 0.55$ and the expected defense reward
is estimated from the EMA attribute on A--D edges.

\section{PageRank Centrality}

Applying PageRank to the \tagraph{} identifies attack types that appear
frequently in attack sequences and are preceded by many diverse attack
types -- a signature of APT pivot attacks:
\begin{equation}
  \mathrm{PR}(A_k) = \frac{1-d}{|V|} + d \sum_{A_{k'} \in \mathcal{N}^-(A_k)}
    \frac{\mathrm{PR}(A_{k'})}{|\mathcal{N}^+(A_{k'})|}
\end{equation}
High PageRank attack nodes are flagged for elevated monitoring by the \arl{}.

\section{Defense Effectiveness Matrix}

The $8 \times 8$ matrix $\mathbf{M}_{\mathrm{def}}$ where
$[\mathbf{M}_{\mathrm{def}}]_{kj} = \E[r_d \mid A_k, D_j]$
provides a compact lookup table for optimal countermeasure selection
when confidence is below $\tau_{\mathrm{pro}}$.
After 7,500 environment events, 64 of the 64 cells are populated,
providing full coverage of the attack-defense space.
